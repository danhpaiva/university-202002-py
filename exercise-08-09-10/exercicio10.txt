#include <Servo.h>

Servo myservo;

int pressionado; //estado do botao
int botao=7; //Pino a ser ligado o botao

void setup()
{
 myservo.attach(9); //pino de dados do servo conectado ao pino 9
 myservo.write(0); //inicia motor posição 0 graus
 pinMode(botao, INPUT_PULLUP);  
}

void loop() 
{
 pressionado=digitalRead(botao);
  
  if(pressionado == HIGH) {
    myservo.write(180);
    delay(500);
  }
}
